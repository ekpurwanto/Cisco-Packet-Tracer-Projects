! ========================================
! MLOCS-SW CONFIGURATION
! Access Layer Switch - MLOCS Department
! Health Services - Project 7
! ========================================

! ========================================
! STEP 1: BASIC SWITCH CONFIGURATION
! ========================================

Switch>enable
Switch#configure terminal

! Set hostname
Switch(config)#hostname MLOCS-SW

! Configure MOTD Banner
MLOCS-SW(config)#banner motd #
************************************************************
*                    WARNING NOTICE                        *
*                                                          *
*   This system is for AUTHORIZED USERS ONLY.              *
*                                                          *
*   Unauthorized access is strictly prohibited.            *
*   All activities are monitored and recorded.             *
*                                                          *
*   Disconnect IMMEDIATELY if you are not authorized.      *
*                                                          *
************************************************************
#

! Disable DNS lookup
MLOCS-SW(config)#no ip domain lookup

! Configure enable password
MLOCS-SW(config)#enable password cisco

! Configure console access
MLOCS-SW(config)#line console 0
MLOCS-SW(config-line)#password cisco
MLOCS-SW(config-line)#login
MLOCS-SW(config-line)#exit

! Encrypt all plaintext passwords
MLOCS-SW(config)#service password-encryption

MLOCS-SW(config)#exit
MLOCS-SW#
%SYS-5-CONFIG_I: Configured from console by console

! Save configuration
MLOCS-SW#write
Building configuration...
[OK]

! ========================================
! STEP 2: TRUNK PORT CONFIGURATION
! ========================================

MLOCS-SW>enable
MLOCS-SW#configure terminal

! Configure trunk ports to Distribution/Core Switch
MLOCS-SW(config)#interface range FastEthernet0/1-2
MLOCS-SW(config-if-range)#switchport mode trunk
MLOCS-SW(config-if-range)#no shutdown
MLOCS-SW(config-if-range)#exit

! ========================================
! STEP 3: ACCESS PORT & VLAN CONFIGURATION
! ========================================

! Configure access ports for MLOCS Department
MLOCS-SW(config)#interface range FastEthernet0/3-24
MLOCS-SW(config-if-range)#switchport mode access
MLOCS-SW(config-if-range)#switchport access vlan 10
MLOCS-SW(config-if-range)#no shutdown
MLOCS-SW(config-if-range)#exit

! Create VLAN 10
MLOCS-SW(config)#vlan 10
MLOCS-SW(config-vlan)#name MLOCS
MLOCS-SW(config-vlan)#exit

! Save configuration
MLOCS-SW(config)#do write
Building configuration...
[OK]

! ========================================
! END OF MLOCS-SW CONFIGURATION
! ========================================
