! ========================================

! Internet Service Provider Router 2

! ========================================
! STEP 1: INTERFACE IP ADDRESSING
! ========================================

ISP-2>enable
ISP-2#configure terminal

! Serial0/2/0 (Connection to HQ-Router)
ISP-2(config)#interface Serial0/2/0
ISP-2(config-if)#ip address 195.136.17.6 255.255.255.252
ISP-2(config-if)#no shutdown
ISP-2(config-if)#exit

! Serial0/2/1 (Connection to Branch-Router)
ISP-2(config)#interface Serial0/2/1
ISP-2(config-if)#ip address 195.136.17.14 255.255.255.252
ISP-2(config-if)#no shutdown
ISP-2(config-if)#exit

ISP-2(config)#do write
Building configuration...
[OK]

! ========================================
! STEP 2: OSPF ROUTING CONFIGURATION
! ========================================

ISP-2#configure terminal
ISP-2(config)#router ospf 10

! Advertise network to HQ-Router
ISP-2(config-router)#network 195.136.17.4 0.0.0.3 area 0

! Advertise network to Branch-Router
ISP-2(config-router)#network 195.136.17.12 0.0.0.3 area 0
ISP-2(config-router)#exit

ISP-2(config)#do write
Building configuration...
[OK]
