! ========================================
! MRM-SW CONFIGURATION
! Access Layer Switch - MRM Department
! Health Services - Project 7
! ========================================

! ========================================
! STEP 1: BASIC SWITCH CONFIGURATION
! ========================================

Switch>enable
Switch#configure terminal

! Set hostname
Switch(config)#hostname MRM-SW

! Configure MOTD Banner
MRM-SW(config)#banner motd #
************************************************************
*                    WARNING NOTICE                        *
*                                                          *
*   This system is for AUTHORIZED USERS ONLY.              *
*                                                          *
*   Unauthorized access is strictly prohibited.            *
*   All activities are monitored and recorded.             *
*                                                          *
*   Disconnect IMMEDIATELY if you are not authorized.      *
*                                                          *
************************************************************
#

! Disable DNS lookup
MRM-SW(config)#no ip domain lookup

! Configure console access
MRM-SW(config)#line console 0
MRM-SW(config-line)#password cisco
MRM-SW(config-line)#login
MRM-SW(config-line)#exit

! Configure enable password
MRM-SW(config)#enable password cisco

! Encrypt all plaintext passwords
MRM-SW(config)#service password-encryption

MRM-SW(config)#exit
MRM-SW#
%SYS-5-CONFIG_I: Configured from console by console

! Save configuration
MRM-SW#write
Building configuration...
[OK]

! ========================================
! STEP 2: TRUNK PORT CONFIGURATION
! ========================================

MRM-SW>enable
MRM-SW#configure terminal

! Configure trunk ports to Distribution/Core Switch
MRM-SW(config)#interface range FastEthernet0/1-2
MRM-SW(config-if-range)#switchport mode trunk
MRM-SW(config-if-range)#no shutdown
MRM-SW(config-if-range)#exit

! ========================================
! STEP 3: ACCESS PORT & VLAN CONFIGURATION
! ========================================

! Configure access ports for MRM Department
MRM-SW(config)#interface range FastEthernet0/3-24
MRM-SW(config-if-range)#switchport mode access
MRM-SW(config-if-range)#switchport access vlan 30
MRM-SW(config-if-range)#no shutdown
MRM-SW(config-if-range)#exit

! Create VLAN 30
MRM-SW(config)#vlan 30
MRM-SW(config-vlan)#name MRM
MRM-SW(config-vlan)#exit

! Save configuration
MRM-SW(config)#do write
Building configuration...
[OK]

! ========================================
! END OF MRM-SW CONFIGURATION
! ========================================
