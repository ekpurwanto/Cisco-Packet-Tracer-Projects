! ========================================
! MK-SW CONFIGURATION
! Access Layer Switch - Marketing (MK) Department
! Melbourne Health Services - Project 7
! ========================================

! ========================================
! STEP 1: BASIC SWITCH CONFIGURATION
! ========================================

Switch>enable
Switch#configure terminal

! Set hostname
Switch(config)#hostname MK-SW

! Configure MOTD Banner
MK-SW(config)#banner motd #
************************************************************
*                    WARNING NOTICE                        *
*                                                          *
*   This system is for AUTHORIZED USERS ONLY.              *
*                                                          *
*   Unauthorized access is strictly prohibited.            *
*   All activities are monitored and recorded.             *
*                                                          *
*   Disconnect IMMEDIATELY if you are not authorized.      *
*                                                          *
************************************************************
#

! Disable DNS lookup
MK-SW(config)#no ip domain lookup

! Configure console access
MK-SW(config)#line console 0
MK-SW(config-line)#password cisco
MK-SW(config-line)#login
MK-SW(config-line)#exit

! Configure enable password
MK-SW(config)#enable password cisco

! Encrypt all plaintext passwords
MK-SW(config)#service password-encryption

MK-SW(config)#exit
MK-SW#
%SYS-5-CONFIG_I: Configured from console by console

! Save configuration
MK-SW#write
Building configuration...
[OK]

! ========================================
! STEP 2: TRUNK PORT CONFIGURATION
! ========================================

MK-SW>enable
MK-SW#configure terminal

! Configure trunk ports to Distribution/Core Switch
MK-SW(config)#interface range FastEthernet0/1-2
MK-SW(config-if-range)#switchport mode trunk
MK-SW(config-if-range)#no shutdown
MK-SW(config-if-range)#exit

! ========================================
! STEP 3: ACCESS PORT & VLAN CONFIGURATION
! ========================================

! Configure access ports for MK Department
MK-SW(config)#interface range FastEthernet0/3-24
MK-SW(config-if-range)#switchport mode access
MK-SW(config-if-range)#switchport access vlan 110
MK-SW(config-if-range)#no shutdown
MK-SW(config-if-range)#exit

! Create VLAN 110
MK-SW(config)#vlan 110
MK-SW(config-vlan)#name MK
MK-SW(config-vlan)#exit

! Save configuration
MK-SW(config)#do write
Building configuration...
[OK]

! ========================================
! END OF MK-SW CONFIGURATION
! ========================================
