! ========================================
! MER-SW CONFIGURATION
! Access Layer Switch - Marketing Department
! Melbourne Health Services - Project 7
! ========================================

! ========================================
! STEP 1: BASIC SWITCH CONFIGURATION
! ========================================

Switch>enable
Switch#configure terminal

! Set hostname
Switch(config)#hostname MER-SW

! Configure MOTD Banner
MER-SW(config)#banner motd #
************************************************************
*                    WARNING NOTICE                        *
*                                                          *
*   This system is for AUTHORIZED USERS ONLY.              *
*                                                          *
*   Unauthorized access is strictly prohibited.            *
*   All activities are monitored and recorded.             *
*                                                          *
*   Disconnect IMMEDIATELY if you are not authorized.      *
*                                                          *
************************************************************
#

! Disable DNS lookup
MER-SW(config)#no ip domain lookup

! Configure console access
MER-SW(config)#line console 0
MER-SW(config-line)#password cisco
MER-SW(config-line)#login
MER-SW(config-line)#exit

! Configure enable password
MER-SW(config)#enable password cisco

! Encrypt all plaintext passwords
MER-SW(config)#service password-encryption

MER-SW(config)#exit
MER-SW#
%SYS-5-CONFIG_I: Configured from console by console

! Save configuration
MER-SW#write
Building configuration...
[OK]

! ========================================
! STEP 2: TRUNK PORT CONFIGURATION
! ========================================

MER-SW>enable
MER-SW#configure terminal

! Configure trunk ports to Distribution/Core Switch
MER-SW(config)#interface range FastEthernet0/1-2
MER-SW(config-if-range)#switchport mode trunk
MER-SW(config-if-range)#no shutdown
MER-SW(config-if-range)#exit

! ========================================
! STEP 3: ACCESS PORT & VLAN CONFIGURATION
! ========================================

! Configure access ports for Marketing Department
MER-SW(config)#interface range FastEthernet0/3-24
MER-SW(config-if-range)#switchport mode access
MER-SW(config-if-range)#switchport access vlan 20
MER-SW(config-if-range)#no shutdown
MER-SW(config-if-range)#exit

! Create VLAN 20
MER-SW(config)#vlan 20
MER-SW(config-vlan)#name MER
MER-SW(config-vlan)#exit

! Save configuration
MER-SW(config)#do write
Building configuration...
[OK]

! ========================================
! END OF MER-SW CONFIGURATION
! ========================================
