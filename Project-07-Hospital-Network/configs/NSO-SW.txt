! ========================================
! NSO-SW CONFIGURATION
! Access Layer Switch - NSO Department
! Melbourne Health Services - Project 7
! ========================================

! ========================================
! STEP 1: BASIC SWITCH CONFIGURATION
! ========================================

Switch>enable
Switch#configure terminal

! Set hostname
Switch(config)#hostname NSO-SW

! Configure MOTD Banner
NSO-SW(config)#banner motd #
************************************************************
*                    WARNING NOTICE                        *
*                                                          *
*   This system is for AUTHORIZED USERS ONLY.              *
*                                                          *
*   Unauthorized access is strictly prohibited.            *
*   All activities are monitored and recorded.             *
*                                                          *
*   Disconnect IMMEDIATELY if you are not authorized.      *
*                                                          *
************************************************************
#

! Disable DNS lookup
NSO-SW(config)#no ip domain lookup

! Configure console access
NSO-SW(config)#line console 0
NSO-SW(config-line)#password cisco
NSO-SW(config-line)#login
NSO-SW(config-line)#exit

! Configure enable password
NSO-SW(config)#enable password cisco

! Encrypt all plaintext passwords
NSO-SW(config)#service password-encryption

NSO-SW(config)#exit
NSO-SW#
%SYS-5-CONFIG_I: Configured from console by console

! Save configuration
NSO-SW#write
Building configuration...
[OK]

! ========================================
! STEP 2: TRUNK PORT CONFIGURATION
! ========================================

NSO-SW>enable
NSO-SW#configure terminal

! Configure trunk ports to Distribution/Core Switch
NSO-SW(config)#interface range FastEthernet0/1-2
NSO-SW(config-if-range)#switchport mode trunk
NSO-SW(config-if-range)#no shutdown
NSO-SW(config-if-range)#exit

! ========================================
! STEP 3: ACCESS PORT & VLAN CONFIGURATION
! ========================================

! Configure access ports for NSO Department
NSO-SW(config)#interface range FastEthernet0/3-24
NSO-SW(config-if-range)#switchport mode access
NSO-SW(config-if-range)#switchport access vlan 80
NSO-SW(config-if-range)#no shutdown
NSO-SW(config-if-range)#exit

! Create VLAN 80
NSO-SW(config)#vlan 80
NSO-SW(config-vlan)#name NSO
NSO-SW(config-vlan)#exit

! Save configuration
NSO-SW(config)#do write
Building configuration...
[OK]

! ========================================
! END OF NSO-SW CONFIGURATION
! ========================================
